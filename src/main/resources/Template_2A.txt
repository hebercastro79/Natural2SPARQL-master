# Template_2A (Workaround 2 - CONTAINS Check)
PREFIX b3: <https://dcm.ffclrp.usp.br/lssb/stock-market-ontology#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT DISTINCT ?individualTicker
WHERE {
  # 1. Encontre a entidade principal pelo label (ex: "GERDAU")
  ?mainEntity rdfs:label #ENTIDADE_NOME# . # Espera uma Label com aspas aqui, ex: "GERDAU"

  # 2. Obtenha a string combinada de tickers ligada a essa entidade principal
  OPTIONAL { ?mainEntity b3:representadoPor/b3:ticker ?combinedTickerString . }

  # 3. Encontre TODOS os recursos de código individuais e seus tickers
  ?anyCode rdf:type b3:Codigo_Negociacao .
  ?anyCode b3:ticker ?individualTicker .

  # --- Estratégia de Filtragem ---
  # Mantenha tickers individuais se eles estão contidos na string combinada (se encontrada)
   FILTER (BOUND(?combinedTickerString) && CONTAINS(?combinedTickerString, ?individualTicker))

  # Filtro Adicional Essencial: Garante formato válido de ticker
  FILTER REGEX(?individualTicker, "^[A-Z]{4}[0-9]{1,2}$")

}
LIMIT 20
